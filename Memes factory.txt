<script>

// CONTROL VISIBILIDAD UI
function hideUIForLogin() {
  document.getElementById("menuBtn").style.display = "none";
  document.getElementById("arrow").style.display = "none";
}

function showUI() {
  document.getElementById("menuBtn").style.display = "block";
  document.getElementById("arrow").style.display = "block";
}

// MENU
function toggleMenu() {
  sidebar.classList.toggle("active");
  overlay.classList.toggle("active");

  if (sidebar.classList.contains("active")) {
    arrow.style.left = "250px";
    arrow.innerHTML = "⬅";
  } else {
    arrow.style.left = "10px";
    arrow.innerHTML = "➡";
  }
}

function closeMenu() {
  sidebar.classList.remove("active");
  overlay.classList.remove("active");
  arrow.style.left = "10px";
  arrow.innerHTML = "➡";
}

// LOGIN
function login() {
  const user = username.value;
  if (!user) return alert("Escribe un nombre");

  localStorage.setItem("user", user);
  initApp();
}

function logout() {
  localStorage.clear();
  location.reload();
}

// SECCIONES
function showSection(id) {
  document.querySelectorAll(".section").forEach(s => s.classList.remove("active"));
  document.getElementById(id).classList.add("active");

  if (id === "gallery") loadGallery();

  closeMenu();
}

// MEME
function updateMeme() {
  topText.innerText = inputTop.value;
  bottomText.innerText = inputBottom.value;
}

// SUBIR IMAGEN
uploadImage.onchange = e => {
  const reader = new FileReader();
  reader.onload = ev => memeImage.src = ev.target.result;
  reader.readAsDataURL(e.target.files[0]);
};

// EXPORTAR
function downloadMeme() {
  html2canvas(meme).then(canvas => {
    const link = document.createElement("a");
    link.download = "meme.png";
    link.href = canvas.toDataURL();
    link.click();
  });
}

// GUARDAR
function saveMeme() {
  if (!memeImage.src) return alert("Sube una imagen");

  let memes = JSON.parse(localStorage.getItem("memes")) || [];

  memes.push({
    img: memeImage.src,
    top: inputTop.value,
    bottom: inputBottom.value
  });

  localStorage.setItem("memes", JSON.stringify(memes));
  alert("Guardado 👍");
}

// GALERÍA
function loadGallery() {
  savedMemes.innerHTML = "";

  let memes = JSON.parse(localStorage.getItem("memes")) || [];

  if (!memes.length) {
    savedMemes.innerHTML = "<p>No tienes memes aún</p>";
    return;
  }

  memes.forEach(m => {
    savedMemes.innerHTML += `
      <img src="${m.img}" width="200"><br>
      <b>${m.top}</b><br>
      <b>${m.bottom}</b><hr>
    `;
  });
}

// INIT
function initApp() {
  if (localStorage.getItem("user")) {
    showUI();
    showSection("create");
  } else {
    hideUIForLogin();
    showSection("login");
  }
}

initApp();
